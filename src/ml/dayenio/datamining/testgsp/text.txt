public String format(String line){
	String out = "";
	bool betweenBrkt = false;
	for(int i = 0; i<line.length(); i++){
		if(line.charAt(i).equals("("))){
			betweenBrkt = true;
			out += line.charAt(i) + " ";
		}else if(line.charAt(i).equals(")"))){
			betweenBrkt = false;
			out += line.charAt(i) + " ";
		}else {
			out += line.charAt(i) + " ";
		}
		if (i==line.length()-1){
			out += "-2";
		}else if (!betweenBrkt){
			out += "-1 "
		}
	}
	
}
